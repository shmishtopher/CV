% Employer macro
% #1 - Employer
% #2 - Location
% #3 - Role
% #4 - Dates
% #5 - List of responsibilities
\newcommand{\employer}[5] {
    \noindent\textbf{#1}\hfill#2\\
    \noindent\textit{#3}\hfill#4
    \begin{compactitem}
        #5
    \end{compactitem}
    \vspace{16pt}
}


% Project macro
% #1 - Name
% #2 - Technologies
% #3 - Link
% #4 - Description

% \noindent#4\\
% \vspace*{50pt}

    \noindent\textbf{#1} --- \textit{#2}\hfill\ifx&#3&\else\href{#3}{#3}\fi\\

}
